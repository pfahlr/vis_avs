cmake_minimum_required(VERSION 3.22)
project(avs-cross-platform VERSION 0.1.0 LANGUAGES C CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

option(ENABLE_SDL3 "Enable experimental SDL3 windowing backend" OFF)
option(ENABLE_PORTAUDIO "Enable PortAudio audio backend" OFF)
option(ENABLE_JACK "Enable JACK audio backend" OFF)
option(ENABLE_FFTW3 "Enable FFTW3 FFT backend" OFF)
option(ENABLE_KISSFFT "Enable kissFFT backend" OFF)
option(ENABLE_FREETYPE "Enable FreeType font rendering" OFF)
option(ENABLE_SDL_TTF "Enable SDL_ttf font rendering" OFF)
option(ENABLE_STB_IMAGE "Enable stb_image image loader" OFF)
option(ENABLE_EEL2 "Enable Cockos WDL EEL2 scripting" OFF)

add_subdirectory(libs)
add_subdirectory(apps)
enable_testing()
add_subdirectory(tests)
